5  Chapter 5 - Data Import/Output and Advanced R

Working with External Data and Advanced R Features

5.1 Learning Objectives

After completing this chapter, you should be able to:

  • Import data from various file formats
  • Read data from web sources and databases
  • Export data in different formats
  • Use advanced data structures
  • Work with dates and times in R

5.2 Overview

This chapter covers material from the lecture slides: Advanced_Statistics_R_4.pdf

5.3 Reading Data with read.table()

The read.table() family of functions provides flexible data import:

# Basic usage
data <- read.table("filename.txt")

# Common options
data <- read.table("filename.txt",
  header = FALSE,          # First line is NOT column names
  sep = "",                # Whitespace delimiter (default)
  dec = ".",               # Decimal point character
  quote = '""',             # Quoting character
  na.strings = "NA",       # String(s) to interpret as NA
  colClasses = NA,         # Column classes (auto-detect)
  nrows = -1,              # Number of rows to read (-1 = all)
  skip = 0,                # Skip first n lines
  check.names = TRUE,      # Make column names valid R names
  fill = FALSE,            # Fill missing values in short rows
  comment.char = "#"       # Character for comments
)

5.3.1 Specialized Functions

# CSV (comma-separated values)
read.csv("file.csv")
read.csv2("file.csv")  # European format (semicolon, comma as decimal)

# Delimited files
read.delim("file.txt")     # Tab-delimited
read.delim2("file.txt")    # Tab-delimited with European decimals

# Fixed width format
read.fwf("file.txt", widths = c(5, 10, 8))

5.4 Importing Data from Various Sources

5.4.1 From the Web

# Read data from URL
url <- "https://raw.githubusercontent.com/.../data.csv"
web_data <- read.csv(url)

# Using httr package for more control
library(httr)
response <- GET("https://api.example.com/data")
content <- content(response, "text")
data <- read.csv(text = content)

5.4.2 From Databases

# Using RSQLite as example
library(DBI)
library(RSQLite)

# Connect to database
con <- dbConnect(RSQLite::SQLite(), "mydb.sqlite")

# List tables
dbListTables(con)

# Read a table
data <- dbReadTable(con, "mytable")

# Execute SQL query
result <- dbGetQuery(con, "SELECT * FROM mytable WHERE age > 25")

# Close connection
dbDisconnect(con)

5.4.3 Binary Data Formats

# HDF5 format (large multidimensional arrays)
library(rhdf5)
h5file <- H5Fopen("data.h5")
data <- h5read(h5file, "dataset_name")
H5Fclose(h5file)

# NetCDF format (climate/weather data)
library(ncdf4)
nc <- nc_open("data.nc")
data <- ncvar_get(nc, "variable_name")
nc_close(nc)

5.4.4 Images

# Using magick package
library(magick)
img <- image_read("path/to/image.png")
print(img)

# For image analysis (pixel values)
library(png)
img_data <- readPNG("image.png")  # Returns numeric array

5.5 Exporting Data

5.5.1 Writing to Files

# Write CSV
write.csv(data, "output.csv", row.names = FALSE)
write.csv(data, "output.csv", na = "", quote = FALSE)

# Write table (more control)
write.table(data, "output.txt",
  sep = "\t",              # Tab-delimited
  quote = FALSE,           # No quotes
  row.names = FALSE,       # No row names
  col.names = TRUE,        # Include column names
  na = "NA"                # NA representation
)

5.5.2 Saving R Objects

# RDS format (single object)
saveRDS(data, "data.rds")
data <- readRDS("data.rds")

# RData format (multiple objects)
save(data1, data2, list_obj, file = "workspace.RData")
load("workspace.RData")  # Loads objects into environment

# Save entire workspace
save.image("my_workspace.RData")

5.7 Set Operations

setA <- c("a", "b", "c", "d", "e")
setB <- c("d", "e", "f", "g")

# Union
union(setA, setB)           # All unique elements
[1] "a" "b" "c" "d" "e" "f" "g"
# Intersection
intersect(setA, setB)       # Common elements
[1] "d" "e"
# Set difference (order matters)
setdiff(setA, setB)         # Elements in A but not in B
[1] "a" "b" "c"
setdiff(setB, setA)         # Elements in B but not in A
[1] "f" "g"
# Set equality
setequal(setA, setB)        # FALSE
[1] FALSE
# Membership
setA %in% setB              # Which elements of A are in B
[1] FALSE FALSE FALSE  TRUE  TRUE
"c" %in% setA               # TRUE
[1] TRUE

5.10 Practical Examples

5.10.1 Example 1: Complete Data Pipeline

# Load and process data
library(tidyverse)

# Read data
data <- read_csv("students.csv", na = c("", "NA", "N/A"))

# Clean and transform
clean_data <- data %>%
  filter(!is.na(age)) %>%
  mutate(
    age_group = cut(age, breaks = c(0, 20, 25, 30, Inf),
                   labels = c("<20", "20-25", "25-30", "30+")),
    gpa_numeric = as.numeric(gsub("[^0-9.]", "", gpa)),
    enrollment_date = as.Date(enrollment_date, format = "%m/%d/%Y")
  ) %>%
  select(-raw_score)  # Remove unnecessary column

# Save cleaned data
write_csv(clean_data, "students_clean.csv")

5.10.2 Example 2: Working with Multiple Files

# Read and combine multiple CSV files
library(tidyverse)

# Get list of files
files <- list.files("data/", pattern = "\\.csv$", full.names = TRUE)

# Read and combine all files
combined_data <- files %>%
  map_dfr(read_csv, .id = "file_id")

# Alternative using base R
combined <- do.call(rbind, lapply(files, read.csv))

5.11 Summary

Key takeaways from this chapter:

  1. Data Import: read.table() family handles most text formats; specialized packages for binary formats
  2. Web Data: Can read directly from URLs; use httr for API access
  3. Databases: DBI package provides unified interface to SQL databases
  4. Dates/Times: Use as.Date() for dates, POSIXct for timestamps, lubridate for easier handling
  5. Tidyverse: Opinionated collection of packages for modern data science
