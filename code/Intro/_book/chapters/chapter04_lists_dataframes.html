4  Chapter 4 - Lists and Data Frames in R

Working with Complex Data Structures

4.1 Learning Objectives

After completing this chapter, you should be able to:

  • Create and manipulate lists in R
  • Build and modify data frames
  • Understand subsetting of data frames
  • Import and export data
  • Work with tibbles (modern data frames)

4.2 Overview

Note

This chapter covers material from the lecture slides: Advanced_Statistics_R_data.pdf

4.3 Lists in R

Lists are Rs most flexible data structure - they can contain elements of different types and lengths.

4.3.1 Creating Lists

# Create a list with different types
my_list <- list(
  name = "Alice",
  age = 25,
  scores = c(85, 90, 78),
  passed = TRUE
)
my_list
$name
[1] "Alice"

$age
[1] 25

$scores
[1] 85 90 78

$passed
[1] TRUE
# Empty list
empty_list <- list()

# List from vector (each element becomes list element)
vec <- 1:5
as.list(vec)
[[1]]
[1] 1

[[2]]
[1] 2

[[3]]
[1] 3

[[4]]
[1] 4

[[5]]
[1] 5

4.3.2 Accessing List Elements

person <- list(name = "Bob", age = 30, scores = c(70, 85, 90))

# Access by index (returns a list)
person[1]           # List with one element
$name
[1] "Bob"
class(person[1])
[1] "list"
# Access by name (returns a list)
person["age"]       # List with one element
$age
[1] 30
# Access with $ (returns the element directly)
person$name         # "Bob"
[1] "Bob"
person$age          # 30
[1] 30
# Access with [[ ]] (returns the element directly)
person[[1]]         # "Bob"
[1] "Bob"
person[["scores"]]  # c(70, 85, 90)
[1] 70 85 90
# Difference between [ ] and [[ ]]
person[1:2]         # List with 2 elements
$name
[1] "Bob"

$age
[1] 30
# person[[1:2]]     # ERROR! [[ ]] accepts only single element

4.3.3 Modifying Lists

# Add new element
person$city <- "New York"
person
$name
[1] "Bob"

$age
[1] 30

$scores
[1] 70 85 90

$city
[1] "New York"
# Modify existing element
person$age <- 31
person[["age"]] <- 32

# Remove element
person$city <- NULL
person
$name
[1] "Bob"

$age
[1] 32

$scores
[1] 70 85 90
# Combine lists
list1 <- list(a = 1, b = 2)
list2 <- list(c = 3, d = 4)
combined <- c(list1, list2)
combined
$a
[1] 1

$b
[1] 2

$c
[1] 3

$d
[1] 4

4.3.4 List Operations

# Length of list
length(person)
[1] 3
# Names of list elements
names(person)
[1] "name"   "age"    "scores"
# Check if element exists
"age" %in% names(person)
[1] TRUE
# Apply function to each element
lapply(person, class)
$name
[1] "character"

$age
[1] "numeric"

$scores
[1] "numeric"
# Recursive unlisting
nested <- list(a = 1, b = list(c = 2, d = 3))
unlist(nested)  # Flattens to vector
  a b.c b.d 
  1   2   3 
unlist(nested, recursive = FALSE)
$a
[1] 1

$b.c
[1] 2

$b.d
[1] 3

4.4 Data Frames in R

Data frames are the workhorse of R data analysis - two-dimensional tables where columns can be different types.

4.4.1 Creating Data Frames

# Create a data frame
df <- data.frame(
  name = c("Alice", "Bob", "Charlie", "Diana"),
  age = c(25, 30, 35, 28),
  score = c(85.5, 92.0, 78.5, 88.0),
  passed = c(TRUE, TRUE, FALSE, TRUE)
)
df

4   Diana  28  88.0   TRUE
# Structure of data frame
str(df)

 $ name  : chr  "Alice" "Bob" "Charlie" "Diana"
 $ age   : num  25 30 35 28
 $ score : num  85.5 92 78.5 88
 $ passed: logi  TRUE TRUE FALSE TRUE
# Summary statistics
summary(df)
     name                age            score         passed       
 Length:4           Min.   :25.00   Min.   :78.50   Mode :logical  
 Class :character   1st Qu.:27.25   1st Qu.:83.75   FALSE:1

                    Mean   :29.50   Mean   :86.00

4.4.2 Accessing Data Frame Elements

# Access columns (all return vectors)
df$age          # Vector
[1] 25 30 35 28
df[["age"]]     # Vector
[1] 25 30 35 28
df[[2]]         # Vector (second column)
[1] 25 30 35 28
df[, 2]         # Vector (second column)
[1] 25 30 35 28
# Access columns (returns data frame)
df["age"]       # Data frame with one column
  age
1  25
2  30
3  35
4  28
df[2]           # Data frame with second column
  age
1  25
2  30
3  35
4  28
df[, 2, drop = FALSE]  # Data frame (preserves structure)
  age
1  25
2  30
3  35
4  28
# Access rows
df[1, ]         # First row (as data frame)
   name age score passed
1 Alice  25  85.5   TRUE
df[1:2, ]       # First two rows
   name age score passed
1 Alice  25  85.5   TRUE
2   Bob  30  92.0   TRUE
# Access specific elements
df[1, 2]        # Row 1, column 2
[1] 25
df$age[1]       # First element of age column
[1] 25
df[1, "age"]    # Row 1, age column
[1] 25

4.4.3 Subsetting Data Frames

# Logical subsetting
df[df$age > 25, ]           # Rows where age > 25
     name age score passed
2     Bob  30  92.0   TRUE
3 Charlie  35  78.5  FALSE
4   Diana  28  88.0   TRUE
df[df$passed == TRUE, ]       # Rows where passed is TRUE
   name age score passed
1 Alice  25  85.5   TRUE
2   Bob  30  92.0   TRUE
4 Diana  28  88.0   TRUE
# Select specific columns
df[, c("name", "score")]     # Only name and score columns
     name score
1   Alice  85.5
2     Bob  92.0
3 Charlie  78.5
4   Diana  88.0
df[1:2, 2:3]                  # First 2 rows, columns 2-3
  age score
1  25  85.5
2  30  92.0
# subset() function
subset(df, age > 25)                     # Rows where age > 25
     name age score passed
2     Bob  30  92.0   TRUE
3 Charlie  35  78.5  FALSE
4   Diana  28  88.0   TRUE
subset(df, age > 25, select = c(name, score))  # Select columns too
     name score
2     Bob  92.0
3 Charlie  78.5
4   Diana  88.0
# Which rows meet condition
which(df$age > 25)           # Returns indices
[1] 2 3 4

4.4.4 Modifying Data Frames

# Add new column
df$grade <- c("B", "A", "C", "B+")
df$passed_10 <- df$score >= 80
df
     name age score passed grade passed_10
1   Alice  25  85.5   TRUE     B      TRUE
2     Bob  30  92.0   TRUE     A      TRUE
3 Charlie  35  78.5  FALSE     C     FALSE
4   Diana  28  88.0   TRUE    B+      TRUE
# Modify existing column
df$age <- df$age + 1  # Everyone is 1 year older

# Remove column
df$grade <- NULL

# Rename columns
names(df)[names(df) == "score"] <- "final_score"
names(df)
[1] "name"        "age"         "final_score" "passed"      "passed_10"  
# Reorder columns
df <- df[, c("name", "age", "final_score", "passed", "passed_10")]
df
     name age final_score passed passed_10
1   Alice  26        85.5   TRUE      TRUE
2     Bob  31        92.0   TRUE      TRUE

4   Diana  29        88.0   TRUE      TRUE

4.5 Importing and Exporting Data

4.5.1 Reading Data

# Read CSV file
data <- read.csv("data.csv")
data <- read.csv("data.csv", header = TRUE, stringsAsFactors = FALSE)

# Read from URL
url_data <- read.csv("https://example.com/data.csv")

# Read with specific options
data <- read.csv("data.csv",
               header = TRUE,          # First row is column names
               sep = ",",              # Field separator
               na.strings = c("NA", ""),  # What to treat as NA
               stringsAsFactors = FALSE)  # Don't convert strings to factors

# Read table (general delimiter)
data <- read.table("data.txt", header = TRUE, sep = "\t")

# Read RDS (R serialized format)
data <- readRDS("data.rds")

# Read Excel (requires readxl package)
# library(readxl)
# data <- read_excel("data.xlsx", sheet = "Sheet1")

4.5.2 Writing Data

# Write to CSV
write.csv(df, "output.csv", row.names = FALSE)
write.csv(df, "output.csv", quote = FALSE, na = "")

# Write to RDS (preserves R objects exactly)
saveRDS(df, "data.rds")

# Save multiple objects
save(df, list, file = "workspace.RData")

# Write to table
write.table(df, "output.txt", sep = "\t", row.names = FALSE)

4.6 The Tidyverse: Tibbles

Tibbles are modern data frames with improved printing and behavior.

4.6.1 Creating Tibbles

library(tibble)

# Create a tibble
df_tibble <- tibble(
  name = c("Alice", "Bob", "Charlie"),
  age = c(25, 30, 35),
  score = c(85.5, 92.0, 78.5)
)
df_tibble
# A tibble: 3 × 3
  name      age score
  <chr>   <dbl> <dbl>
1 Alice      25  85.5
2 Bob        30  92  
3 Charlie    35  78.5
# Convert data frame to tibble
df_as_tibble <- as_tibble(df)
df_as_tibble
# A tibble: 4 × 5
  name      age final_score passed passed_10
  <chr>   <dbl>       <dbl> <lgl>  <lgl>    
1 Alice      26        85.5 TRUE   TRUE     
2 Bob        31        92   TRUE   TRUE     
3 Charlie    36        78.5 FALSE  FALSE    
4 Diana      29        88   TRUE   TRUE     
# tribble: transposed tibble (row-wise entry)
df_tribble <- tribble(
  ~name,    ~age, ~score,
  "Alice",   25,   85.5,
  "Bob",     30,   92.0,
  "Charlie", 35,   78.5
)
df_tribble
# A tibble: 3 × 3
  name      age score
  <chr>   <dbl> <dbl>
1 Alice      25  85.5
2 Bob        30  92  
3 Charlie    35  78.5

4.6.2 Tibble Features

# Tibbles don't convert strings to factors
class(df_tibble$name)  # "character"
[1] "character"
# Tibbles have nicer printing
# Only shows first 10 rows
# Shows column types
# Doesn't print row names

# Tibbles are stricter about subsetting
df_tibble$name         # Works (returns vector)
[1] "Alice"   "Bob"     "Charlie"
df_tibble[, "age"]    # Returns tibble (not vector!)
# A tibble: 3 × 1
    age
  <dbl>
1    25
2    30
3    35
df_tibble[["age"]]     # Returns vector
[1] 25 30 35

4.6.3 List Columns in Tibbles

# Tibbles can have list columns (advanced feature)
df_with_list <- tibble(
  id = 1:3,
  name = c("A", "B", "C"),
  measurements = list(
    c(1.2, 2.3, 3.4),
    c(2.1, 3.2),
    c(4.5, 5.6, 6.7, 7.8)
  )
)
df_with_list
# A tibble: 3 × 3

  <int> <chr> <list>      
1     1 A     <dbl [3]>   
2     2 B     <dbl [2]>   
3     3 C     <dbl [4]>   
# Access list column
df_with_list$measurements[[1]]  # First person's measurements

4.7 The data.table Package

For large datasets, data.table offers faster operations.

library(data.table)

# Create data.table
dt <- data.table(
  name = c("Alice", "Bob", "Charlie"),
  age = c(25, 30, 35),
  score = c(85.5, 92.0, 78.5)
)
dt
      name   age score
    <char> <num> <num>
1:   Alice    25  85.5
2:     Bob    30  92.0
3: Charlie    35  78.5
# Fast subsetting
dt[age > 25]              # Filter rows
      name   age score
    <char> <num> <num>
1:     Bob    30  92.0
2: Charlie    35  78.5
dt[, .(name, score)]      # Select columns
      name score
    <char> <num>
1:   Alice  85.5
2:     Bob  92.0
3: Charlie  78.5
dt[, mean_score := mean(score)]  # Add column
dt[, .(avg_age = mean(age)), by = score > 80]  # Group by
    score avg_age
   <lgcl>   <num>
1:   TRUE    27.5
2:  FALSE    35.0
Tidyverse vs data.table
  • Tidyverse (dplyr/tibble): Readable code, great for analysis
  • data.table: Fast performance, memory efficient
  • Both are excellent; choose based on your needs

4.8.2 Example 2: Data Cleaning

# Handle missing values
df_with_na <- data.frame(
  a = c(1, 2, NA, 4),
  b = c(5, NA, 7, 8),
  c = c(9, 10, 11, 12)
)

# Remove rows with any NA
df_no_na <- na.omit(df_with_na)
df_no_na
  a b  c
1 1 5  9
4 4 8 12
# Replace NAs with specific value
df_filled <- df_with_na
df_filled$a[is.na(df_filled$a)] <- mean(df_filled$a, na.rm = TRUE)

4.9 Summary

Key takeaways from this chapter:

  1. Lists: Flexible containers for mixed data types; use [[ ]] or $ to access elements
  2. Data Frames: Two-dimensional data structures; columns can be different types
  3. Subsetting: Use [row, column] syntax; logical conditions for filtering
  4. Import/Export: read.csv/write.csv for common formats; saveRDS for R objects
  5. Tibbles: Modern data frames with better printing and stricter behavior
